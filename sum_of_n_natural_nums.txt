.data
input: .asciiz "Enter limit for the sum of natural number = "
show: .asciiz " \n sum of natural numbers are =   "

.text
main:
#get the integer input from user
    li $v0, 4
    la $a0, input
    syscall
    li $v0, 5
    syscall
    move $s5, $v0
    li $v0, 1
    move $a0, $s5
    syscall
    
    addi $t5, $t5, 0
iteration:
    blt $t5, $s5, increment 
    j end
increment:  
    add $s2, $s2, $t5
    addi $t5, $t5, 1
j iteration
end:    
    li $v0, 4
    la $a0, show
    syscall  
    
    li $v0, 1
    move $a0, $s2
    syscall
